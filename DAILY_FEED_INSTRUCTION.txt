ARTICLE SCHEMA

Each article JSON object must contain:

  tab        : "games" | "industry" | "art"  (lowercase English ID)
  headline   : Korean, 25-45 chars, no period at end, lead with subject
  summary    : Korean, 60-110 chars, news-wire tone, ends in 다/한다/했다/이다/밝혔다
  body       : Korean HTML string, 600-800 Korean chars total. Structure:
               (1) THREE SHORT OPENING PARAGRAPHS in casual news-feed tone (친근한 말투):
                     Para 1: 무슨 일이 있었는지
                     Para 2: 구체적인 내용이나 배경
                     Para 3: 왜 흥미롭거나 중요한지
               (2) TWO TO THREE BODY PARAGRAPHS — concise Korean translation/summary
                   of the article's substantive content. Trim navigation, ads,
                   author bios, and related-article boxes.
               (3) ATTRIBUTION — final paragraph:
                   <p>원문: <a href="ORIGINAL_URL">SOURCE_NAME</a></p>
               Wrap every paragraph in <p>...</p>. Use double quotes for HTML
               attributes. No \n line breaks inside the string.
  source     : Outlet name. Korean outlets in Korean, English outlets in English.
  hue        : Integer 0-360. games: try 20-50 or 300-340. industry: 200-240.
               art: 260-290 or 40-80.
  url        : Primary source URL with https://
  urls       : OPTIONAL array of {label, href}. Korean labels: 관련 보도, 공식 발표,
               영상, 깃허브, 논문, 보도자료. OMIT entirely if no extra links.
  image      : OPTIONAL og:image URL. OMIT entirely if no image found.
               Do NOT write image: "" or image: null.

python3 << 'PYEOF'
import subprocess, re

sources = [
    "https://www.cgchannel.com/feed/",
    "https://80.lv/feed/",
    "https://www.gamedeveloper.com/rss.xml",
    "https://www.marktechpost.com/feed/",
    # Add others as needed
]

candidates = []
for url in sources:
    r = subprocess.run(
        ['curl', '-s', '-L', '--max-time', '8',
         '-A', 'Mozilla/5.0 (compatible; Googlebot/2.1)', url],
        capture_output=True, text=True, timeout=12
    )
    xml = r.stdout
    items = re.findall(
        r'<item>.*?<title><!\[CDATA\[(.*?)\]\]></title>.*?<link>(.*?)</link>.*?<pubDate>(.*?)</pubDate>.*?</item>',
        xml, re.DOTALL
    )
    for title, link, pub in items[:10]:
        candidates.append((pub.strip(), link.strip(), title.strip()))

for pub, link, title in candidates:
    print(f"{pub[:22]} | {title[:60]} | {link[:80]}")

Step C — From all candidate URLs collected in Steps A and B, batch-check
publication dates in a single script. Only keep articles from the last 24 hours:

python3 << 'PYEOF'
import subprocess, re
from datetime import date, timedelta

today = date.today()
yesterday = today - timedelta(days=1)
cutoff_dates = {today.strftime('%Y-%m-%d'), yesterday.strftime('%Y-%m-%d')}

# Fill this list with ALL candidate URLs from Steps A and B
candidates = [
    "https://example.com/article-1",
    "https://example.com/article-2",
    # ... all candidates
]

confirmed = []
for url in candidates:
    r = subprocess.run(
        ['curl', '-s', '-L', '--max-time', '8',
         '-A', 'Mozilla/5.0 (compatible; Googlebot/2.1)', url],
        capture_output=True, text=True, timeout=12
    )
    html = r.stdout
    date_m = re.search(
        r'(?:datePublished|publishedAt|article:published_time)["\s:]+["\']?(\d{4}-\d{2}-\d{2})',
        html, re.IGNORECASE
    )
    pub_date = date_m.group(1) if date_m else 'UNKNOWN'
    status = '✅ CONFIRMED' if pub_date in cutoff_dates else f'❌ SKIP ({pub_date})'
    print(f"{status} | {url[:80]}")
    if pub_date in cutoff_dates:
        confirmed.append(url)

print(f"\n{len(confirmed)} confirmed articles within 24 hours.")
for u in confirmed:
    print(f"  {u}")

STEP A — Fetch the article content and extract image + extra links:

python3 << 'PYEOF'
import subprocess, re
from urllib.parse import urljoin, urlparse

url = "REPLACE_WITH_CONFIRMED_URL"
result = subprocess.run(
    ['curl', '-s', '-L', '--max-time', '8',
     '-H', 'User-Agent: Mozilla/5.0 (compatible; Googlebot/2.1)', url],
    capture_output=True, text=True, timeout=12
)
html = result.stdout

# Extract og:image
image_url = ''
for p in [
    r'property=["\']og:image["\'][^>]*content=["\']([^"\']+)["\']',
    r'content=["\']([^"\']+)["\'][^>]*property=["\']og:image["\']',
    r'name=["\']twitter:image["\'][^>]*content=["\']([^"\']+)["\']',
    r'content=["\']([^"\']+)["\'][^>]*name=["\']twitter:image["\']',
    r'<article[^>]*>.*?<img[^>]+src=["\']([^"\']+)["\']',
]:
    m = re.search(p, html, re.IGNORECASE | re.DOTALL)
    if m:
        image_url = m.group(1).strip()
        break

# Extract candidate extra links
hrefs = re.findall(r'<a[^>]+href=["\']([^"\'#][^"\']*)["\']', html, re.IGNORECASE)
base = urlparse(url).netloc
links, seen = [], set()
for href in hrefs:
    full = href if href.startswith('http') else urljoin(url, href)
    p = urlparse(full)
    if full not in seen and p.netloc and p.netloc != base and (
        any(v in p.netloc for v in ['youtube.com','youtu.be','vimeo.com']) or
        any(k in full.lower() for k in ['github.com','huggingface.co','demo',
            'trial','product','features','release','launch','app','arxiv.org'])
    ):
        seen.add(full); links.append(full)
    if len(links) >= 8: break

# Extract visible article body text
clean = re.sub(r'<(script|style|nav|footer|header|aside|iframe|form)[^>]*>.*?</\1>',
               '', html, flags=re.DOTALL|re.IGNORECASE)
clean = re.sub(r'<[^>]+>', ' ', clean)
clean = re.sub(r'[ \t]+', ' ', clean)
clean = re.sub(r'\n\s*\n+', '\n', clean).strip()
article_text = clean[:6000]

print("OG_IMAGE:", image_url or "NO_IMAGE")
for l in links:
    print("LINK:", l)
print("---ARTICLE_TEXT---")
print(article_text)

python3 << 'PYEOF'
import json, re, os, hashlib
from datetime import date

REPO = '/home/user/Daily_Art_AI_News_Feed'
ARTICLES_JS = os.path.join(REPO, 'articles.js')

today = date.today()
upload_date = today.strftime('%Y.%m.%d')
slug_suffix = today.strftime('%Y-%m')

with open('/tmp/feed/articles.json') as f:
    new_articles = json.load(f)

with open(ARTICLES_JS, 'r', encoding='utf-8') as f:
    original = f.read()

existing_ids = set(re.findall(r"id:\s*'([^']+)'", original))

def make_slug(headline, tab, idx):
    base = f"{tab}-{hashlib.md5((headline + str(idx)).encode()).hexdigest()[:8]}"
    candidate = f"{base}-{slug_suffix}"
    n = 2
    while candidate in existing_ids:
        candidate = f"{base}-{n}-{slug_suffix}"
        n += 1
    existing_ids.add(candidate)
    return candidate

def js_string(s):
    s = s.replace('\\', '\\\\')
    s = s.replace("'", "\\'")
    return "'" + s + "'"

def render_article(art, idx):
    aid = make_slug(art['headline'], art['tab'], idx)
    lines = ['        {']
    lines.append(f"            id:          {js_string(aid)},")
    lines.append(f"            tab:         {js_string(art['tab'])},")
    lines.append(f"            headline:    {js_string(art['headline'])},")
    lines.append(f"            summary:     {js_string(art['summary'])},")
    lines.append(f"            body:        {js_string(art['body'])},")
    lines.append(f"            source:      {js_string(art['source'])},")
    lines.append(f"            publishedAt: {js_string(upload_date)},")
    lines.append(f"            isNew:       true,")
    lines.append(f"            hue:         {int(art['hue'])},")
    if 'image' in art and art['image']:
        lines.append(f"            image:       {js_string(art['image'])},")
    lines.append(f"            url:         {js_string(art['url'])},")
    if 'urls' in art and art['urls']:
        lines.append(f"            urls: [")
        for lk in art['urls']:
            lines.append(f"                {{ label: {js_string(lk['label'])}, href: {js_string(lk['href'])} }},")
        lines.append(f"            ],")
    lines.append('        },')
    return '\n'.join(lines)

by_tab = {'games': [], 'industry': [], 'art': []}
for i, art in enumerate(new_articles):
    by_tab[art['tab']].append((i, art))

SECTION_MARKERS = {
    'games':    '/* ---- games ---- */',
    'industry': '/* ---- industry ---- */',
    'art':      '/* ---- art ---- */',
}

# Strip isNew from old articles, then insert today's new ones
stripped = re.sub(r'^\s*isNew:\s*true,\s*\n', '', original, flags=re.MULTILINE)
updated = stripped

for tab, items in by_tab.items():
    if not items:
        continue
    marker = SECTION_MARKERS[tab]
    if marker not in updated:
        raise RuntimeError(f"Section marker not found: {marker}")
    block = '\n'.join(render_article(a, i) for i, a in items)
    updated = updated.replace(marker, marker + '\n' + block, 1)

with open(ARTICLES_JS, 'w', encoding='utf-8') as f:
    f.write(updated)

added = sum(len(v) for v in by_tab.values())
print(f"Updated articles.js: +{added} new articles.")
print(f"  games:    +{len(by_tab['games'])}")
print(f"  industry: +{len(by_tab['industry'])}")
print(f"  art:      +{len(by_tab['art'])}")

Step 3 — Push via token URL (local proxy denies pushes). Retry with backoff.

python3 << 'PYEOF'
import subprocess, time, os

REPO = '/home/user/Daily_Art_AI_News_Feed'
token = os.environ['GITHUB_TOKEN']
remote_url = f'https://{token}@github.com/kyoungwhankim/Daily_Art_AI_News_Feed.git'
branch = 'main'

push_env = os.environ.copy()
push_env['GIT_TRACE'] = '1'

MAX_RETRIES = 4
delays = [2, 4, 8, 16]

for attempt in range(1, MAX_RETRIES + 1):
    print(f'Push attempt {attempt}/{MAX_RETRIES}...', flush=True)
    r = subprocess.run(
        ['git', 'push', '--progress', '--verbose', remote_url, branch],
        cwd=REPO, capture_output=True, text=True, env=push_env, timeout=120,
    )
    if r.returncode == 0:
        print('✅ Pushed.')
        print(r.stderr.replace(token, '***').strip()[-300:])
        break
    print('Attempt failed:')
    print(r.stderr.replace(token, '***')[-300:])
    if attempt < MAX_RETRIES:
        time.sleep(delays[attempt - 1])
else:
    raise SystemExit('❌ All push attempts failed.')
